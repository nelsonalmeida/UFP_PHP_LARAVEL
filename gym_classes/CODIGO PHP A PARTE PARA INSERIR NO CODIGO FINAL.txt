<?php
function search_athlete_change(){
  $id_athlete = $_POST["ID_Search_change"];

  $id_athletes = DB::select('select * from users where id = ?', [$id_athlete]);

  echo"<table border=1>";
  echo "<tr><td>NOME</td><td>EMAIL</td><td>PASSWORD</td><td>ADMIN</td></tr>";

  foreach ($id_athletes as $athlete){
	  echo "<tr><td>"$athlete->name"</td><td>".$athlete->email."</td><td>".$athlete->password."</td><td>".$athlete->admin."</td></tr>";
  }
  echo"</table>";

}

?>
